cmake_minimum_required(VERSION 3.16)
project(CPP_Rendering_Engine)

set(CMAKE_CXX_STANDARD 20)

# In a new 'build' folder type:
# cmake .. -DCMAKE_TOOLCHAIN_FILE=/home/vcpkg/scripts/buildsystems/vcpkg.cmake
# cmake --build

if(NOT DEFINED CMAKE_TOOLCHAIN_FILE)
    set(CMAKE_TOOLCHAIN_FILE "/home/vcpkg/scripts/buildsystems/vcpkg.cmake" CACHE STRING "")
endif()
set(CMAKE_SKIP_INSTALL_RULES ON)
set(CMAKE_SKIP_RPATH ON)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -static-libgcc -static-libstdc++ -static")



find_package(SDL2 CONFIG REQUIRED)
find_package(glm CONFIG REQUIRED)

add_executable(TestSDL2GLM ./src/core/main.cpp)
target_link_libraries(TestSDL2GLM PRIVATE SDL2::SDL2)

if(WIN32)
    set_target_properties(TestSDL2GLM PROPERTIES LINK_FLAGS "-mwindows")
    target_link_libraries(TestSDL2GLM PRIVATE SDL2::SDL2main SDL2::SDL2)
endif()
